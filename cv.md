function isWalkValid(walkArr) {
  let x = 0
  let y = 0
  walkArr.forEach(walk => {
    if (walk === 'e') x += 1
    if (walk === 'w') x -= 1
    if (walk === 'n') y += 1

  })
  return ((x === 0) && (y === 0) && (walkArr.length === 10))
}
